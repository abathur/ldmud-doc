@usage{
foo = ({ 1, 2, 3, 4 });         str = "test";

foo[1]     -> 1                 str[1] -> 'e' == 101
foo[1..2]  -> ({ 2, 3 })        str[1..2]  -> "es"
foo[2..1]  -> ({ })             str[2..1]  -> ""
foo[0..<2] -> ({ 1, 2 })        str[0..<2]  -> "tes"
foo[..<2]  -> ({ 1, 2 })        str[..<2]  -> "tes"
foo[<3..]  -> ({ 2, 3, 4 })     str[<3..]  -> "est"

foo[1] = 5                -> foo == ({ 1, 5, 3, 4 })
foo[1..2] = ({ 5, 6, 7 }) -> foo == ({ 1, 5, 6, 7, 4 })
foo[1..2] = ({ })         -> foo == ({ 1, 4 })

str[1] = 'a'              -> str == "tast"
str[1..2] = "bar"         -> str == "tbart"
str[1..2] = ""            -> str == "tt"
}
