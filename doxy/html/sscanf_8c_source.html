2  /**
3  * Parse a string <str> using the format <fmt>. <fmt> can contain
4  * strings seperated by %d and %s. Every %d and %s corresponds to
5  * one of <var1>, <var2>, ... .
6  *
7  * The match operators in the format string have one of these
8  * formats:
9  *
10  * %[+][!|~][<size>[.<minmatch>]]<type>
11  *
12  * <type> may be:
13  * d: matches any number.
14  * D: matches any number.
15  * U: matches any unsigned number.
16  * s: matches any string.
17  * %: matches the % character.
18  * t: matches whitespace (spaces and tab characters), but does
19  *
20  * not store them (the simple ' ' matches just spaces and
21  * can't be given a size specification).
22  *
23  * <size> is the expected field size, and <minmatch> the demanded
24  * minimal match length (defaults are 0 for strings and 1 for
25  * numbers). Each of these both may be specified numerically, or
26  * as '*' - in which case the value of the variable at the current
27  * place in the argument list is used.
28  *
29  * Specifying + will require that the characters after the field
30  * match as well, or the match will be deemed unsuccessful (the variable
31  * might still get assigned, though).
32  *
33  * Specifying ! will perform the match, but neither store the
34  * result nor count the match.
35  *
36  * Specifying ~ will perform and count the match, but not store
37  * the result.
38  *
39  * If the %s specifier is not at the end of the format string,
40  * it is matched only if the following character(s) or format
41  * is found, too. See below for an example.
42  *
43  * The difference between %d and %D/%U is that the latter will abort
44  * an immediately preceeding %s as soon as possible, whereas the
45  * former will attempt to make the largest match to %s first.
46  * %D/%U will still not skip whitespace; use %.0t%D to skip optional
47  * whitespace.
48  *
49  * The number of matched arguments will be returned.
50  *
51  * The function sscanf is special in that arguments are passed
52  * by reference automatically.
53  *
54  * @synopsis{
55  * int sscanf(string str, string fmt, mixed var1, mixed var2, ...)
56  * }
57  *
58  * @usage{
59  * string who, what;
60  * if (sscanf("throw frisbee to rover",
61  * "throw %s to %s", what, who) != 2)
62  * write("Usage: throw <what> to <who>\n");
63  * else
64  * write("You throw a "+what+" to "+who+" to get his attention.\n");
65  *
66  * sscanf("ab", "%s%s", who, what)
67  * ==> result 2, who = "", what = "ab"
68  *
69  * sscanf("ab", "%s %s", who, what)
70  * ==> result 0, who = 0, what = 0
71  *
72  * sscanf("ab ", "%s %s", who, what)
73  * ==> result 2, who = "ab", what = ""
74  * }
75  *
76  **/
77  int sscanf(string str, string fmt, mixed var1, mixed var2, ...);
78 }
