2  /**
3  * This efun is for backward compatibility only. It is only available in compat mode.
4  *
5  * Move the object \p item to the object \p dest. All kinds of tests are done, and a number is returned specifying the result:
6  *
7  * 0: Success.
8  * 1: Too heavy for destination.
9  * 2: Can't be dropped.
10  * 3: Can't take it out of it's container.
11  * 4: The object can't be inserted into bags etc.
12  * 5: The destination doesn't allow insertions of objects.
13  * 6: The object can't be picked up.
14  *
15  * If an object is transfered to a newly created object, make sure that the new object first is transfered to its destination.
16  *
17  * The efun calls applied::add_weight(), applied::drop(), applied::get(), applied::prevent_insert(), applied::add_weight(), and applied::can_put_and_get() where needed.
18  *
19  * ## replacement simul_efun #
20  *
21  * This efun can easily be replaced with a simul_efun:
22  *
23  * ~~~{.c}
24  * /*--------------------------------------------------------*/
25  * int transfer(object item, object dest)
26  * {
27  * int weight;
28  * object from;
29  *
30  * efun::set_this_object(previous_object());
31  *
32  * weight = item->query_weight();
33  * if (!item)
34  * return 3;
35  *
36  * from = environment(item);
37  * if (from)
38  * {
39  * /*
40  * * If the original place of the object is a living object,
41  * * then we must call drop() to check that the object can be
42  * * dropped.
43  * */
44  * if (living(from))
45  * {
46  * if (item->drop() || !item)
47  * return 2;
48  * }
49  * /*
50  * * If 'from' is not a room and not a player, check that we may
51  * * remove things out of it.
52  * */
53  * else if (environment(from))
54  * {
55  * if (!from->can_put_and_get() || !from)
56  * return 3;
57  * }
58  * }
59  *
60  * /*
61  * * If the destination is not a room, and not a player,
62  * * Then we must test 'prevent_insert', and 'can_put_and_get'.
63  * */
64  * if (environment(dest) && !living(dest))
65  * {
66  * if (item->prevent_insert())
67  * return 4;
68  * if (!dest->can_put_and_get() || !dest)
69  * return 5;
70  * }
71  *
72  * if (living(dest))
73  * {
74  * if (!item->get() || !item)
75  * return 6;
76  * }
77  *
78  * /*
79  * * If it is not a room, correct the total weight in the
80  * * destination.
81  * */
82  * if (environment(dest) && weight)
83  * {
84  * if (!dest->add_weight(weight) || !dest)
85  * return 1;
86  * }
87  *
88  * /*
89  * * If it is not a room, correct the weight in the 'from' object.
90  * */
91  * if (from && environment(from) && weight)
92  * {
93  * from->add_weight(-weight);
94  * }
95  *
96  * move_object(item, dest);
97  *
98  * return 0;
99  * }
100  * /*--------------------------------------------------------*/
101  *
102  * ~~~
103  *
104  * @synopsis{
105  * int transfer(object item, object dest)
106  * }
107  *
108  * @history{
109  * deprecated (3.3) -- available only when compiled with USE_DEPRECATED.
110  * }
111  *
112  * @see efun::move_object(), applied::drop(), applied::get(), applied::prevent_insert(), applied::can_put_and_get(), applied::add_weight()
113  **/
114  int transfer(object item, object dest);
115 }
