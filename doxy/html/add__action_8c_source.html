2  /**
3  * Set up a local function fun to be called when user input
4  * matches the command cmd. Functions called by a user command
5  * will get the arguments as a string. It must then return 0 if
6  * it was the wrong command, otherwise 1.
7  *
8  * If it was the wrong command, the parser will continue
9  * searching for another command, until one returns 1 or give an
10  * error message to the user.
11  *
12  * For example, there can be a wand and a rod. Both of these
13  * objects define as command "wave". One of them will be randomly
14  * called first, and it must look at the argument, and match
15  * against "wand" or "rod" respectively.
16  *
17  * The function associated to a command will be called with a
18  * string as argument which stands for the given words behind the
19  * typed command. The verb entered can be retrieved using the
20  * query_verb() efun and is always the first word in the input
21  * line up to the first space.
22  *
23  * Always have add_action() called only from an init() routine.
24  * The object defining these commands must be present to the
25  * user, either being the user, being carried by the user,
26  * being the room around the user, or being an object in the
27  * same room as the user. If the player leaves this vicinity of the
28  * object, the actions are automatically removed.
29  *
30  * Actions can also be removed on demand with the remove_actions() efun.
31  *
32  * If argument <flag> is AA_SHORT (1), then the arguments may
33  * follow the verb without separating space. Any arguments after
34  * the first space are passed as argument string.
35  *
36  * If argument <flag> is AA_NOSPACE (2), then again the arguments
37  * may follow the verb without separating space. In contrast to
38  * AA_SHORT, all characters following the verb are passed as
39  * the argument string. However, note that the characters immediately
40  * following the given verb are passed as argument AND as result
41  * of query_verb().
42  *
43  * If argument <flag> is AA_IMM_ARGS (3), then again the arguments
44  * may follow the verb without separating space. All characters following
45  * the given verb are passed as argument, and only as argument.
46  *
47  * If argument <flag> is negative, the verb given by the player
48  * has to match only the leading -<flag> characters of the verb <cmd>.
49  *
50  * Never use one of the functions 'create' 'reset' 'init' 'exit'
51  * 'heart_beat' etc as the first argument to add_action(). In
52  * general, a function with a name defined in /doc/applied should
53  * have the behaviour defined there.
54  *
55  * @headerfile <commands.h>
56  *
57  * @synopsis{
58  * void add_action(string|closure fun, string cmd)
59  * void add_action(string|closure fun, string cmd, int flag)
60  * }
61  *
62  * @usage{
63  * add_action("GoInside", "enter");
64  * When typing "enter" the function GoInside() will be invoked.
65  * add_action("DisFunc", "dis", AA_SHORT);
66  * Whenever you type in a command which starts with "dis" the
67  * function DisFunc() will be called. To get the real word which
68  * was typed in (because until now you only know that it was a
69  * command beginning with "dis") you have to call the efun
70  * query_verb().
71  * add_action("DisFunc", "disconnect", AA_NOSPACE);
72  * The function DisFunc() will be called for all commands which
73  * use "disconnect" or a shortened form like "di", "dis" or
74  * "discon" as verb. The command 'disconnecting' will _not_
75  * be recognized. To get the real word which was typed in
76  * you have to call the efun query_verb().
77  * add_action("...", "cmd");
78  * add_action("...", "xcmd", AA_SHORT);
79  * add_action("...", "scmd", AA_NOSPACE);
80  * add_action("...", "icmd", AA_IMM_ARGS);
81  * When given the following commands, the driver will parse it
82  * as described below. 'verb' is what query_verb() would return,
83  * 'arg' is what would be passed to the command function.
84  *
85  * "cmd" -> verb "cmd", arg 0
86  * "cmd foo bar" -> verb "cmd", arg "foo bar"
87  *
88  * "xcmd" -> verb "xcmd", arg 0
89  * "xcmd foo" -> verb "xcmd", arg "foo"
90  * "xcmdfoo" -> verb "xcmdfoo", arg 0
91  * "xcmd foo bar" -> verb "xcmd", arg "foo bar"
92  * "xcmdfoo bar" -> verb "xcmdfoo", arg "bar"
93  *
94  * "scmd" -> verb "scmd", arg 0
95  * "scmd foo" -> verb "scmd", arg " foo"
96  * "scmdfoo" -> verb "scmdfoo", arg "foo"
97  * "scmd foo bar" -> verb "scmd", arg " foo bar"
98  * "scmdfoo bar" -> verb "scmdfoo", arg "foo bar"
99  *
100  * "icmd" -> verb "icmd", arg 0
101  * "icmd foo" -> verb "icmd", arg " foo"
102  * "icmdfoo" -> verb "icmd", arg "foo"
103  * "icmd foo bar" -> verb "icmd", arg " foo bar"
104  * "icmdfoo bar" -> verb "icmd", arg "foo bar"
105  * }
106  *
107  **/
108  void add_action(string|closure fun, string cmd, int flag);
109 }
