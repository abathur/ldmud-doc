2  /**
3  * Most of the characters in the format string \p fmt get passed straight through to the output (i.e.: printed or put in the return string), to format arguments into the string it is necessary to include an argument format string (AFS) in the format string. An AFS is a series of characters starting with a percent sign "%" and terminated with a argument type specifier.
4  *
5  * To include a "%" sign in the output, it is necessary to include a double percent sign "%%". The sequence "%^" will output "%^" again.
6  *
7  * Valid argument type specifiers are:
8  * "s" : the argument is a string.
9  * "d" : the argument is an integer to be included in decimal representation.
10  * "i" : same as "d".
11  * "b" : the argument is an integer to be included in binary representation.
12  * "o" : the argument is an integer to be included in octal representation.
13  * "x" : the argument is an integer to be included in hexadecimal representation.
14  * "X" : as "x" except letters are capitalised.
15  * "e","E","f","g","G" : the argument is a float to be included in decimal representation; see examples for details.
16  * "c" : the argument is an int to included as a character.
17  * "O" : the argument is an LPC datatype to be printed in an arbituary format, this is for debugging purposes. If the argument is an object the function master::printf_obj_name() is called on the master object with the object as a parameter; the string returned is included in brackets at the end of object file name. If 0 is returned, nothing is appended after the file name.
18  * "Q" Like "O", except that special characters in strings are printed in LPC notation.
19  *
20  * Between the percent sign and the argument type specifier in the AFS, the following modifiers can be included to specify the formatting information. Order is not important unless otherwise specified. "n" is used to specify a integer, which can be a "*" in which case the next argument is used as the number.
21  *
22  * Modifiers:
23  * n specifies the field size. If the size is prepended with a 0, the argument is printed with leading zeroes.
24  *
25  * "."n specifies the precision, for simple strings (not columns or tables) specifies the truncation length.
26  * ":"n n specifies both the field size _and_ the precision; if n is prepended by a zero then the pad string is set to "0".
27  * "'X'" the pad string is set to the char(s) between the single quotes; if the field size is also prepended with a zero then whichever is specified last will overrule. NOTE: to include "'" in the pad string, you must use "\'" (as the backslash has to be escaped past the interpreter); similarly, to include "" you must use "\\".
28  * " " pad positive integers with a space.
29  * "+" pad positive integers with a plus sign.
30  * "-" left-aligned within field size.
31  *
32  * NB: std (s)printf() defaults to right alignment, which is unnatural in the context of a mainly string based language but has been retained for "compatibility" ;)
33  *
34  * "|" centered within field size.
35  * "$" justified to field size. Ignored unless the type specifier is 's'.
36  * "=" column mode. Ignored unless the argument type specifier is 's'. Field size must be specified; if precision is specified then it sets the width for the string to be wordwrapped at, but otherwise the field size is used. Field size specifies the width of the column the last line of the column is padded with spaces to achieve this length.
37  * "#" For strings: table mode.
38  *
39  * Field size must be specified; if precision is specified it sets the number of columns in the table, otherwise the number is "optimally" generated (as few lines and columns as possible). Table mode is passed a list of backslash-n separated 'words' which are put in a format similar to that of ls.
40  *
41  * For %O/%Q: compact output.
42  *
43  * "@" the argument is an array. The corresponding AFS (minus all "@") is applied to each element of the array.
44  *
45  * When the formatting of an element results in several output lines (column or table mode) and no explicit pad string has been defined, any padding whitespace before the newlines of all but the last line is removed. However, if an explicit pad string has been given, even if it is the simple char:` `(U+00020 SPACE), the padding will not be removed.
46  *
47  * @synopsis{
48  * string sprintf(string fmt, ...)
49  * }
50  *
51  * @usage{
52  * ~~~{.c}
53  * sprintf("decimal=%d, octal=%o, hexadecimal=%x\n", 7, 7, 7);
54  *
55  * sprintf("array=%O\n", ({1, 2, 3}));
56  *
57  * // this will return the following:
58  * // ({ // sizeof() == 3
59  * // 1,
60  * // 2,
61  * // 3
62  * // })
63  * // An array will be printed recursively and each element of the
64  * // array will be indented. Can also be used as a debugging tool.
65  *
66  * sprintf("%-*#s\n", 80, implode(get_dir("~/."), "\n"));
67  *
68  * sprintf("foo") // returns "foo"
69  *
70  * sprintf("%s","foo") // returns "foo"
71  * sprintf("%7s","foo") // returns " foo"
72  * sprintf("%-7s","foo") // returns "foo "
73  * sprintf("%|7s","foo") // returns " foo "
74  * sprintf("%7'.'s","foo") // returns "....foo"
75  * sprintf("%-7'+-'s","foo") // returns "foo+-+-"
76  * sprintf("%|9'-+'s","foo") // returns "-+-foo-+-"
77  * sprintf("%3s","foobarbloh") // returns "foobarbloh"
78  * sprintf("%3.6s","foobarbloh") // returns "foobar"
79  * sprintf("%6.3s","foobarbloh") // returns " foo"
80  * sprintf("%:6s","foobarbloh") // returns "foobar"
81  * sprintf("%:3s","foobarbloh") // returns "foo"
82  * sprintf("%*.*s",-7,2,"foobarbloh") // returns "fo "
83  *
84  * sprintf("%=12s","this is a very long sentence\n")
85  * // returns " this is a\n"
86  * // " very long\n"
87  * // " sentence\n"
88  * sprintf("%=-12s","this is a very long sentence\n")
89  * // returns "this is a\n"
90  * // "very long\n"
91  * // "sentence\n"
92  * sprintf("%=|12s","this is a very long sentence\n")
93  * // returns " this is a\n"
94  * // " very long\n"
95  * // " sentence\n"
96  * sprintf("%=10.6s","this is a very long sentence\n")
97  * // returns " this\n"
98  * // " is a\n"
99  * // " very\n"
100  * // " long\n"
101  * // " senten\n"
102  * // " ce\n"
103  * sprintf("%#-40.3s",
104  * "one\ntwo\nthree\nfour\nfive\nsix\nseven\neight\nnine\nten\n")
105  * // returns "one five nine\n"
106  * // "two six ten\n"
107  * // "three seven \n"
108  * // "four eight "
109  * sprintf("%#-40s",
110  * "one\ntwo\nthree\nfour\nfive\nsix\nseven\neight\nnine\nten\n")
111  * // returns "one three five seven nine\n"
112  * // "two four six eight ten"
113  *
114  * sprintf("%@-5s",({"foo","bar","bloh"})) // returns "foo bar bloh "
115  *
116  * sprintf("%d",123) // returns "123"
117  * sprintf("%7d",123) // returns " 123"
118  * sprintf("%-7d",123) // returns "123" (yes, really)
119  * sprintf("%d/%d",123,-123) // returns "123/-123"
120  * sprintf("% d/% d",123,-123) // returns " 123/-123"
121  * sprintf("%+d/%+d",123,-123) // returns "+123/-123"
122  * sprintf("%+6d/%6d",123,123) // returns " +123/ 123"
123  * sprintf("%|6d",123) // returns " 123" (yes, really)
124  * sprintf("%|10d",123) // returns " 123" (yes, really)
125  * sprintf("%|10d%3s",123,"foo") // returns " 123 foo"
126  *
127  * sprintf("%o",16) // returns "20"
128  * sprintf("%'0'3o",8) // returns "010"
129  *
130  * sprintf("%x",123) // returns "7b"
131  * sprintf("%X",123) // returns "7B"
132  *
133  * sprintf("%f",123.5) // returns "124"
134  * sprintf("%8.3f",123.5) // returns " 123.500"
135  * sprintf("%E",123.5) // returns "1E+02"
136  * sprintf("%12.4e",123.5) // returns " 1.2350e+02"
137  * sprintf("%g",123.5) // returns "124"
138  * sprintf("%8.3G",123.5) // returns " 124"
139  * sprintf("%8.6g",123.5) // returns " 123.5"
140  *
141  * ~~~
142  *
143  * }
